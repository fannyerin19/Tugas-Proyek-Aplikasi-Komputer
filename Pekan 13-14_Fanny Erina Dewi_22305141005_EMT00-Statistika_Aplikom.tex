\begin{eulercomment}

secara berurutan.
\end{eulercomment}

kontinu X dikatakan memiliki distribusi eksponensial, jika PDF-nya

     ind=nonzeros(f==i);

\begin{eulerprompt}
>function readtd (line) ...
\end{eulerprompt}
\begin{eulerudf}
  v=[]; cp=0;
  repeat
     \{pos,s,vt\}=strxfind(line,"<td.*?>(.+?)</td>",cp);
     until pos==0;
     if length(vt)>0 then v=v|vt[1]; endif;
     cp=pos+strlen(s);
  end;
  return v;
  endfunction
\end{eulerudf}
